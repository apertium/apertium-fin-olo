    Macros

    Macros are helper functions in apertium transfer files.

    remove-semtag1

    Parametres: 1
    1. let tl[1]['a_sem'] ≔ ""

    tensemood-mangler

    Parametres: 1
      1. if
        1. sl[1]['a_tense'] ≟ <past>
        3. then:
        1. let tl[1]['a_tense'] ≔ <ind.prt>
      3. elseif
        1. sl[1]['a_tense'] ≟ <pri>
        5. then:
        1. let tl[1]['a_tense'] ≔ <ind.prs>

    Rules

    The actual rules concerning stuff.

    REGLA: COND

    Matching pattern:

    1. cond

    Action:

    1. let tl[1]['a_sem'] ≔ ""
    2. let tl[1]['x_func'] ≔ ""
    3. let tl[1]['a_func'] ≔ ""
    4. let tl[1]['a_cond'] ≔ <cond.prs>
    5. Output:
      1. [
        1. tl[1]['whole']
        ]cond sl[1]['a_func']

    REGLA: COND QST

    Matching pattern:

    1. cond
    2. qst

    Action:

    1. let tl[1]['a_sem'] ≔ ""
    2. let tl[1]['x_func'] ≔ ""
    3. let tl[1]['a_func'] ≔ ""
    4. let tl[1]['a_cond'] ≔ <cond.prs>
    5. Output:
      1. [
          ]cond sl[1]['a_func']

      REGLA: CONNEG

      Matching pattern:

      1. conneg

      Action:

      1. let tl[1]['a_sem'] ≔ ""
      2. let tl[1]['x_func'] ≔ ""
      3. let tl[1]['a_func'] ≔ ""
        1. if
          1. tl[1]['a_nbr'] ≟ <pl>
          3. then:
          1. let tl[1]['a_nbr'] ≔ ""
          2. let tl[1]['tags'] ≔ tl[1]['tags']<p3.pl>
        1. if
          1. tl[1]['a_temps'] ≟ <prs>
          3. then:
          1. let tl[1]['a_temps'] ≔ ""
      6. Output:
        1. [
          1. tl[1]['whole']
          ]conneg sl[1]['a_func']

      REGLA: pron

      Matching pattern:

      1. pron

      Action:

      1. remove-semtag1($1)
      2. let tl[1]['x_func'] ≔ ""
      3. let tl[1]['a_func'] ≔ ""
      4. let tl[1]['a_poss'] ≔ ""
        1. if
          1. tl[1]['a_cas'] ≟ <acc>
          3. then:
          1. let tl[1]['a_cas'] ≔ <par>
      6. Output:
        1. [
          1. tl[1]['whole']
          ]pron <SN>

      REGLA: nom

      Matching pattern:

      1. nom

      Action:

      1. remove-semtag1($1)
      2. let tl[1]['x_func'] ≔ ""
      3. let tl[1]['a_func'] ≔ ""
      4. let tl[1]['a_poss'] ≔ ""
      5. Output:
        1. [
          1. tl[1]['whole']
          ]nom <SN>

      REGLA: nom kaa

      Matching pattern:

      1. nom
      2. kanssa

      Action:

      1. remove-semtag1($1)
      2. let tl[1]['x_func'] ≔ ""
      3. let tl[1]['a_func'] ≔ ""
      4. let tl[1]['a_poss'] ≔ ""
      5. let tl[1]['a_cas'] ≔ <com>
      6. Output:
        1. [
          1. tl[1]['whole']
          ]nom <SN>

      REGLA: inf verb

      Matching pattern:

      1. inf

      Action:

      1. remove-semtag1($1)
      2. let tl[1]['x_func'] ≔ ""
      3. let tl[1]['a_func'] ≔ ""
      4. let tl[1]['a_actv'] ≔ ""
      5. let tl[1]['a_infa'] ≔ <inf>
      6. let tl[1]['a_cas'] ≔ ""
      7. let tl[1]['a_nbr'] ≔ ""
      8. Output:
        1. [
          1. tl[1]['whole']
          ]v <SV>

      REGLA: verb qst

      Matching pattern:

      1. verb
      2. qst

      Action:

      1. remove-semtag1($1)
      2. let tl[1]['x_func'] ≔ ""
      3. let tl[1]['a_func'] ≔ ""
      4. let tl[1]['a_act'] ≔ ""
      5. Output:
        1. [
            ]v <SV>

        REGLA: verb

        Matching pattern:

        1. verb

        Action:

        1. remove-semtag1($1)
        2. tensemood-mangler($1)
        3. let tl[1]['x_func'] ≔ ""
        4. let tl[1]['a_func'] ≔ ""
        5. let tl[1]['a_act'] ≔ ""
        6. Output:
          1. [
            1. tl[1]['lem'] tl[1]['a_verb'] tl[1]['a_actv'] tl[1]['a_tense'] tl[1]['a_pers'] tl[1]['a_nbr']
            ]v <SV>

        REGLA: ADV

        Matching pattern:

        1. adv

        Action:

        1. let tl[1]['a_sem'] ≔ ""
        2. let tl[1]['x_func'] ≔ ""
        3. let tl[1]['a_func'] ≔ ""
        4. let tl[1]['a_poss'] ≔ ""
        5. Output:
          1. [
            1. tl[1]['whole']
            ]default sl[1]['a_func']

        REGLA: DEFAULT

        Matching pattern:

        1. default

        Action:

        1. let tl[1]['a_sem'] ≔ ""
        2. let tl[1]['x_func'] ≔ ""
        3. let tl[1]['a_func'] ≔ ""
        4. Output:
          1. [
            1. tl[1]['whole']
            ]default sl[1]['a_func']

        REGLA: DEFAULT QST

        Matching pattern:

        1. default
        2. qst

        Action:

        1. let tl[1]['a_sem'] ≔ ""
        2. let tl[1]['x_func'] ≔ ""
        3. let tl[1]['a_func'] ≔ ""
        4. Output:
          1. [
              ]default sl[1]['a_func']
